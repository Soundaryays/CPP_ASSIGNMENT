Write an algorithm to check whether a given number is a prime number. In addition, write a program to implement this algorithm.

 

Hint 1: A number is a prime number if it is divisible only by one and itself.

Hint 2: Check the divisibility of the number by dividing it by 2, 3, â€¦.., n/2. If it is divisible by any of these numbers, it cannot be a prime number. However, if it is not divisible by any of these numbers, it is definitely a prime number.

#include <iostream>
using namespace std;

int main() 
{
    int i, n;
    bool Prime = true;

    cout << "Enter a number: ";
    cin >> n;

    // 0 and 1 are not prime numbers
    if (n == 0 || n == 1) {
       Prime=false;
      
    }
    else {
        for (i = 2; i <= n / 2; i++) {
            if (n % i == 0) {
               Prime = false;
            }
        }
    }
    if (Prime)
        cout << n << " is a prime number";
    else
        cout << n << " is not a prime number";

    return 0;
}
